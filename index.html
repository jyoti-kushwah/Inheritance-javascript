Q2-Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays?

Q3-Write code to explain prototype chaining?

 
        function Person() {
        this.name = 'John'
    }
    
    // adding property 
    Person.prototype.name = 'Peter';
    Person.prototype.age = 23
    
    const person1 = new Person();
    
    console.log(person1.name); // John
    console.log(person1.age); // 23
